debug.js - debug message demo

HTML:
include the script file.
<script src="debug.js"></script>

set a debug messag area.
<div id="debug"></div>

JavaScript:
output the debug message.
debug(<message>);

(optional)
If you want to change the log settings, explicitly call initialization function in advance.
arguments: (<id of the message area>, <buffer size>, <show line numbers>)
Debug.init("debug", 20, true);

e.g.,)
<!DOCTYPE html>
<html>
<head>
  <meta charset="utf-8">
  <script src="debug.js"></script>
  <script>
    window.onload = function() {
      Debug.init('debug', 20, true); //in the absence of this statement, debug.js works in default configuration.
      debug('The window\'s load event fired.');
    }

    document.onkeydown = function(e) {
      debug('KeyDown: keyCode = ' + e.keyCode);
    }

    document.onmousemove = function(e) {
      debug('mouseMoved: x=' + e.clientX + ', y=' + e.clientY);
    };
    ...
  </script>
</head>
<body>
  Hello, World!<br>
  ...
  <div id="debug"></div>
  ...
</body>
</html>
