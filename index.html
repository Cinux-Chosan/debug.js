debug.js - demo & description
HTML:
include the script file.
<script src="debug.js"></script>

JavaScript:
put the log messages anywhere you want.
log(<message>);

That's it!

The debug window is draggable and you can switch hide/show by F2 key.

e.g.,)
<!DOCTYPE html>
<html>
<head>
  <meta charset="utf-8">
  <script src="debug.js"></script>
  <script>
    window.onload = function() {
      log('The window\'s load event fired.');
    }

    document.onkeydown = function(e) {
      log('KeyDown: keyCode = ' + e.keyCode);
    }

    document.onmousemove = function(e) {
      log('mouseMoved: x=' + e.clientX + ', y=' + e.clientY);
    };
    ...
  </script>
</head>
<body>
  Hello, World!<br>
  ...
</body>
</html>
(optional)
If you want to change the log settings, explicitly call initialization function in advance.
var options = {
  'buffSize': 18,
  'width': 500,
  'height': 215,
  'position': 'right-bottom', // left-top, left-bottom, center, right-top, right-bottom
  'posAdjX': 20,
  'posAdjY': 20,
  'errorColor': '#d44',
  'warnColor': '#ed0',
  'infoColor': '#fff',
  'debugColor': '#8cf',
  'verboseColor': '#ccc',
  'specialColor': '#fff',
  'timeColor': '#0f0',
  'systemInfoColor': '#ddd',
  'showLineNums': true,
  'showTimeStamp': true,
  'showClock': true,
  'showClearButton': true,
  'showCloseButton': true,
  'showWindowSize': true,
  'showMouseStatus': true,
  'showKeyStatus': true,
  'enableStopWatch': true,
  'enableCommandLine': true
};

Debug.init(null, options);
You can also use the log function with log type.
log.e('error log');
log.w('warn log');
log.i('info log');
log.d('debug log');
log.v('verbose log');
log.s('special log');
