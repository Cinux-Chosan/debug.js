debug.js
Easy Debugging w/o F12
Debugger of JavaScript, by JavaScript, for JavaScript
Logging, Time Test, Element Inspection, Screen Measure, and more
// Usage

Embed the debugger into your web page
Include the script file. 
<script src="debug.js"></script>
That's it!

The debug window will be automatically generated and is draggable.

⚠The debug window is hidden by default setting.
You can switch show/hide by F2 key.

Logging
Put log messages anywhere you want.
log(message);
e.g.,)
<!DOCTYPE html>
<html>
<head>
  <meta charset="utf-8">
  <script src="debug.js"></script>
  <script>
    function foo() {
      log('button was clicked');
    }
  </script>
</head>
<body>
  <button onclick="foo();">BUTTON</button> 
</body>
</html>

Print an object
You can also print all the properties of an object.
function bar() {
  var obj = {
    'key1': 'value1',
    'key2': 123,
    'key3': true
  };
  log.p(obj); 
}

HTTP Request
var request = {
  'url': '',
  'method': '',
  'data': '',
  'async': ,
  'cache': ,
  'user': '',
  'pass': '',
  'userAgent': ''
};

dbg.httpRequest(request, httpCallback); 

function httpCallback(xhr) {
  var statusMsg = xhr.status + ' ' + xhr.statusText;
  if (xhr.status == 0) {
    log.e('cannot load: ' + statusMsg);
  } else {
    log(statusMsg);
  }
  var head = xhr.getAllResponseHeaders();
  var txt = xhr.responseText.replace(/</g, '&lt;');
  txt = txt.replace(/>/g, '&gt;');
  if (head || txt) {
    log(head + '\n'+ txt);
  }
}
